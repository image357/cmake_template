project(docs)

        REQUIRED dot
        OPTIONAL_COMPONENTS mscgen dia
)

        include/${CMAKE_PROJECT_NAME}
        ALL
        WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
)
